      import axios from 'axios';

      mounted() {
      this.fetchTickets();

      methods: {
      fetchTickets() {

      });

      calculateCounts() {
      this.categoryCounts = this.tickets.reduce((acc, ticket) => {

      }
      return acc;

      this.statusCounts = this.tickets.reduce((acc, ticket) => {

      }
