<template>
<v-card height="100%" tile>
  <v-toolbar flat>
    <v-toolbar-title><v-icon>mdi-shield-plus</v-icon> Salidas </v-toolbar-title>
    <v-spacer />
  </v-toolbar>
  <v-divider />
  <v-card-text class="pa-1">
    <v-container class="pa-2" fluid>
      <v-row dense>
        <v-col cols="12">
          <v-card>
            <v-card-title class="pa-2">
              <v-container class="pa-0" fluid>
                <v-row dense>
                  <v-btn @click="dialog = true" class="ma-4" outlined small icon color="primary"><v-icon>mdi-plus</v-icon></v-btn>
                  <v-spacer />
                  <v-col cols="12" sm="4">
                    <searchs
                      :outlined="true"
                      :iheaders-search="headersSearch"
                      @searchState="searchState"
                      @refrescar="refrescar"
                      @buscar="buscar"
                    />
                  </v-col>
                  <exits
                    :edited-id="editedId"
                    :edited-index="editedIndex"
                    :dialog="dialog"
                    @closedialog="closedialog"
                    @exitEdit="buscar"
                    @exitChange="buscar"
                  />
                </v-row>
              </v-container>
            </v-card-title>
            <v-divider />
            <v-card-text class="pa-0">
              <v-data-table
                :mobile-breakpoint="0"
                :footer-props="variableGlobals.rowsPerPageItems"
                :headers="headers"
                :items="exit"
                :loading="loading"
                :options.sync="pagination"
                :items-per-page="itemsPerPage"
                :page.sync="paginaActual"
                :loading-text="'Cargando... Espere por favor'"
                dense
                :server-items-length="total"
                no-data-text="No existen registros"
              >
                <template #[`item.index`]="{ item }">
                  <strong> {{ exit.indexOf(item) + 1 }}.</strong>
                </template>
                <template #[`item.ruc`]="{ item }">
                  <span> {{ item.ruc }}</span>
                </template>
                <template #[`item.actions`]="{ item }">
                  <actions :edit-show="true" :delete-show="true" :register="item" @editRegister="editExit" @deleteRegister="deleteExit" />
                </template>
              </v-data-table>
            </v-card-text>
          </v-card>
        </v-col>
      </v-row>
    </v-container>
  </v-card-text>
</v-card>
</template>

<script>
import axios from 'axios'
import config from '~/config'
import variableGlobals from '~/plugins/variables'
import exits from '~/components/administracion/addExit'
import actions from '~/components/globals/actions'
import searchs from '~/components/globals/search'
export default {
name: 'Empleados',
components: {
  exits,
  actions,
  searchs
},
middleware: ['auth', 'permission'],
async asyncData ({ store }) {
  try {
    const exit = await axios.get(`${config.URL}exits`)
    return { exit: exit.data.data, total: exit.data.docs }
  } catch (error) {
    // if (error.response && error.response.status !== 404) error({ statusCode: error.response.status, messagePublic: error.response.data.messagePublic })
  }
},
data () {
  return {
    dialog: false,
    editedId: '',
    pagination: {
      sortBy: '',
      page: 1,
      itemsPerPage: 15
    },
    editedIndex: -1,
    variableGlobals: variableGlobals,
    query: '',
    loading: false,
    total: 0,
    headers: [
      { text: 'NÂ°', value: 'index', sortable: false },
      { text: 'Cliente', value: 'clientName', align: 'rigth', sortable: false },
      { text: 'Salida', value: 'amount', align: 'rigth', sortable: false },
      { text: 'Acciones', value: 'actions', align: 'center', sortable: false }
    ],
    headersSearch: [
      { text: 'Nombres y Apellidos', value: 'ExitName' }
    ],
    structureOr: '',
    itemsPerPage: 15,
    paginaActual: 1,
    exit: [],
  }
},
watch: {
  pagination: {
    handler (val, oldVal) {
      if (val.page !== oldVal.page || val.itemsPerPage !== oldVal.itemsPerPage) this.buscar()
    },
    deep: true
  }
},
 beforeMount () {
  axios.get(`${config.URL}exits`).then((data) => {
  this.exit = data.data.data
  console.log(this.exit);
   })
},
methods: {
  async buscar () {
    this.loading = true
    let query = this.structureOr
    let params = {
      skip: this.pagination.page - 1,
      limit: this.pagination.itemsPerPage,
      query: query
    }
    await axios.get(`${config.URL}exits`, {
      headers: this.$store.getters['auth/getAuthHeader'](this.$store.getters['permission/getHeaders']),
      params: {
          query: query
        }
    }).then((data) => {
      this.exit = data.data.data
      console.log(this.exit, 'data')
      this.loading = false
    }).catch((error) => {
      this.loading = false
      console.log(error)
    })
    this.loading = false
  },
  deleteExit (item) {
    axios.delete(`${config.URL}exit/${item._id}`, { headers: this.$store.getters['auth/getAuthHeader'](this.$store.getters['permission/getHeaders']) }).then((data) => {
      this.buscar()
    }).catch((error) => {
      console.log(error)
      this.$root.$emit('NotificationError', error.response.data.message)
    })
  },
  changeStatus (item) {
    axios.put(`${config.URL}statusExit/${item._id}`, { status: !item.status }, { headers: this.$store.getters['auth/getAuthHeader'](this.$store.getters['permission/getHeaders']) }).then((data) => {
      this.buscar()
    }).catch((error) => {
      console.log(error)
      this.$root.$emit('NotificationError', error.response.data.message)
    })
  },
  refrescar (structureOr) {
    this.structureOr = structureOr
    this.buscar()
  },
  searchState (search, structureOr) {
    this.structureOr = structureOr
    this.pagination.page = 1
  },
  editExit(exit) {
    console.log(exit, 'pagina')
    this.editedIndex = this.exit.indexOf(exit)
    this.editedId = exit._id
    this.dialog = true
  },
  closedialog () {
    this.dialog = false
    setTimeout(() => {
      this.editedIndex = -1
      this.editedId = ''
    }, 500)
  }
}
}
</script>
