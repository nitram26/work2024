<template>
<v-text-field
outlined
dense
hide-details
:value="correlativeWithPrefix"
@input="updateCorrelative" />
</template>

<script>
export default {
data() {
return {
boxsmall: {
correlative: ''
}
};
},
computed: {
correlativeWithPrefix() {
return this.boxsmall.correlative ? this.boxsmall.correlative : 'CA-';
}
},
methods: {
updateCorrelative(value) {
// Asegúrate de que siempre comience con "CA-"
if (!value.startsWith('CA-')) {
value = `CA-${value}`;
}
this.boxsmall.correlative = value;
}
}
};
</script>
