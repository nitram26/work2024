fech (fi) {
  const date = new Date(fi)
  const year = date.getFullYear()
  let month = date.getMonth()
  let dt = date.getDate()
  month++
  if (dt < 10) {
    dt = '0' + dt
  }
  if (month < 10) {
    month = '0' + month
  }
  return `${dt}/${month}/${year}`

    formatDate(dateString) {
      // Crea una nueva fecha a partir del string
      const date = new Date(dateString);
      // Obtén los componentes de la fecha
      const year = date.getFullYear();
      const month = ('0' + (date.getMonth() + 1)).slice(-2); // Asegura dos dígitos
      const day = ('0' + date.getDate()).slice(-2); // Asegura dos dígitos
      // Devuelve la fecha en formato deseado
      return `${year}-${month}-${day}`;
    },
  },
