Props

  1. Crear el view y el componente
  2. Padre: Se crean las variables, se importa el hijo y se envían las variables en el template
  3. Hijo: recibe los valores en el script con el uso de defineProps
  4. Se muestran en el template del view
  5. La combinación de padre-hijo
          

          
componente Padre

          <template>
  <v-container :fluid="true">
   <h1>Padre</h1> 

   <WorkView :nombre="nombre" :edad="edad"/>
  </v-container>
</template>
<!-- ******************************* --> 
<script setup>
import { ref } from 'vue'
import WorkView from './WorkView.vue'

let nombre = ref('Carlos')
let edad = ref(65)
</script>
<!-- ******************************* -->
<style scoped>

</style>

////////////////////////////////////////////////////

componente hijo

<template>
  <div>
   <h1>Hijo</h1> 
   <h2>Te llamas {{ nombre }} y tienes {{ edad }} años</h2>
  </div>
</template>
<!-- ******************************* --> 
<script setup>

const props = defineProps({
  nombre: String,
  edad: Number
})
</script>
<!-- ******************************* -->

</style>
