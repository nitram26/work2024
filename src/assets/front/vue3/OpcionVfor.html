Opciones V-FOR 0745

  1. hacer un card y tener un objeto de preguntas
  2. que cada pregunta con su respuesta aparezcan
  3. 5 preguntas con 3-4 respuestas
  4. Que desaparezcan después de 3 segundos las respuestas
    

      <template>
      <div>
          <h2>Opciones con V-for</h2>
          <div class="card" v-for="(titulo, index) in preguntas" :key="index" >
            <h2 >{{ titulo.pregunta }}</h2>
            <ol>
              <li v-for="(respuesta, index) in titulo.opciones" :key="index" class="res">{{ respuesta }}</li>
              <h2 :class="color">{{ titulo.respuestaCorrecta }}</h2>
            </ol>      
            <button @click="rpta" >RESPUESTA</button>
    
          </div>
      </div>
    </template>
    <!-- ************************************* -->
    <script setup>
    import { ref } from 'vue';
    
    let color = ref('rojo')
    const rpta = () => {
    color.value = 'verde'
    
    const restablecerColor = () => {
        color.value = 'rojo';
    }
    setTimeout(restablecerColor, 3000)
    }
    const preguntas = [
      { 
          pregunta: "¿Lenguaje más utilizado?",
          opciones: ["JavaScript", "Python", "Java", "C++"],
          respuestaCorrecta: "JavaScript"
      },
      {
          pregunta: "¿Significado de CSS?",
          opciones: ["Cascading Style Sheets", "Creative Style Selector", "Code Styling System", "Crazy Stylish Syntax"],
          respuestaCorrecta: "Cascading Style Sheets"
      },
      {
          pregunta: "¿Qué tipo de lenguaje es Python?",
          opciones: ["Interpretado", "Compilado", "Ambos", "Ninguno de los anteriores"],
          respuestaCorrecta: "Interpretado"
      },
      {
          pregunta: "¿Qué es HTML?",
          opciones: ["Un lenguaje de programación", "Un sistema de gestión de bases de datos", "Un lenguaje de marcado", "Un lenguaje de consulta"],
          respuestaCorrecta: "Un lenguaje de marcado"
      },
      {
          pregunta: "¿Cuál de los siguientes no es un lenguaje de programación?",
          opciones: ["HTML", "CSS", "Java", "Ruby"],
          respuestaCorrecta: "CSS"
      },
      
    ]
    
    </script>
    <!-- ************************************* -->
    <style scoped>
    .card{
    background-color: darkcyan;
    width: 30em;
    height: 17em;
    border: greenyellow 2px solid;
    border-radius: 1em;
    margin: 1em;
    padding: 0.8em;
    
    }
    .rojo{
    color: darkcyan;
    }
    .verde{
    color:white
    }
    
    .res{
      margin-left: 2em;
    }
    </style>
