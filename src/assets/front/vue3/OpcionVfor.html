Opciones V-FOR

  1. hacer un card y tener un objeto de preguntas
  2. que cada pregunta con su respuesta aparezcan
  3. 5 preguntas con 3-4 respuestas
  4. Que desaparezcan después de 3 segundos las respuestas
    

      <template>
  <h1>Practica  </h1>

  <div class="div" v-for="(preg, index) in preguntasComputacion" :key="index">
   <h5> {{ preg.pregunta }}</h5>
   <ol>
    <li v-for="(rpta, index) in preg.opciones" :key="index">{{ rpta }}</li> <br><br>
   <h2 v-show="ver">La respuesta es  {{ preg.respuesta }} </h2>
    <v-btn class="bg-teal" @click="res">RESPUESTA</v-btn>
   </ol>
  </div>

</template>
<!-- ************************************* -->
<script setup>
import { ref } from 'vue';


let ver = ref(false)
let res = ()=>{
  ver.value = !ver.value
  setTimeout(() => {
    ver.value = !ver.value
  }, 2000);
}


const preguntasComputacion = [
        {
            pregunta: "¿Cuál de los siguientes es un lenguaje de programación orientado a objetos?",
            opciones: [
                "A) COBOL",
                "B) C++",
                "C) Assembly",
                "D) Fortran"
            ],
            respuesta: "B) C++"
        },
        {
            pregunta: "¿Qué es una base de datos relacional?",
            opciones: [
                "A) Un sistema para gestionar archivos de texto",
                "B) Un sistema que utiliza tablas para almacenar datos y establecer relaciones entre ellos",
                "C) Un programa para crear gráficos y diagramas",
                "D) Un lenguaje de programación utilizado para desarrollar aplicaciones web"
            ],
            respuesta: "B) Un sistema que utiliza tablas para almacenar datos y establecer relaciones entre ellos"
        },
        {
            pregunta: "¿Qué representa 'HTTP' en el contexto de la web?",
            opciones: [
                "A) HyperText Transfer Protocol",
                "B) HyperText Transmission Program",
                "C) HyperText Transfer Program",
                "D) HyperText Transmission Protocol"
            ],
            respuesta: "A) HyperText Transfer Protocol"
        },
        {
            pregunta: "¿Cuál de los siguientes NO es un sistema operativo?",
            opciones: [
                "A) Windows",
                "B) Linux",
                "C) Python",
                "D) macOS"
            ],
            respuesta: "C) Python"
        },
        {
            pregunta: "¿Qué significa la sigla 'SQL' en el contexto de bases de datos?",
            opciones: [
                "A) Structured Query Language",
                "B) Simple Query Language",
                "C) Sequential Query Language",
                "D) Structured Question Language"
            ],
            respuesta: "A) Structured Query Language"
        }]
</script>
