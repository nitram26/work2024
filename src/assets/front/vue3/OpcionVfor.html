Opciones V-FOR 1745

  1. hacer un card y tener un objeto de preguntas
  2. que cada pregunta con su respuesta aparezcan
  3. 5 preguntas con 3-4 respuestas
  4. Que desaparezcan después de 3 segundos las respuestas
    

      <template>
  <h1>PRACTICA JULIO 2024</h1>
  
 <div v-for="(preg, index) in preguntasComputacion" :key="index" class="card">
 <span>{{ index+1 }}.- {{ preg.pregunta }}</span>
 <ul>
  <li v-for="(option, index) in preg.opciones" :key="index">{{ option }}</li> <br><br>
</ul>
<h2 v-show="ver">{{ preg.respuesta }}</h2>
<v-btn @click="see">Push</v-btn>
</div>


</template>
<!-- ************************************* -->
<script setup>
import { ref } from 'vue';

let ver = ref(false)

let see = ()=>{
  ver.value = !ver.value

  let color = ()=>{
    ver.value = !ver.value
  }
  setTimeout(color, 3000)
}

  const preguntasComputacion = [
