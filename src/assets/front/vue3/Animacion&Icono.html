    <script setup>
    import { ref } from 'vue';

Transiciones a Grupos

  1. se hace un agregar elementos normal
    2.        <template>
     <div class="container">
       <ol >
         <li v-for="(note, index) in notes" :key="index">{{ note }}</li>
   
       </ol> <br><br>
       <input type="text" v-model="note" class="bg-purple">
       <br><br>
       <v-btn class="bg-teal" @click="handleNote">Save / Note</v-btn>
     </div>
   
   </template>
   <!-- ************************************* -->
   <script setup>
   import { ref } from 'vue';
   
   
   const notes = ref([])
   const note = ref('')
   
   const handleNote = ()=>{
     notes.value.push(note.value)
     note.value = ''
   }
   </script>
   <!-- ************************************* -->
   <style scoped>
   
   .container{
     margin: 2em;
     padding: 2em;
     border: brown 4px double;
   }
   </style>

TransitionGroup

  • Para hacer la transicion de grupos se cambia el elemeto padre
    en este caso el ol por transitionGroup
    •     <template>
    <div class="container">
      <TransitionGroup name="list" tag="ol">
        <li v-for="(note, index) in notes" :key="index">{{ note }}</li>
  
      </TransitionGroup> <br><br>
      <input type="text" v-model="note" class="bg-purple">
      <br><br>
      <v-btn class="bg-teal" @click="handleNote">Save / Note</v-btn>
    </div>
  
  </template>
  <!-- ************************************* -->
  <script setup>
  import { ref } from 'vue';
  
  
  const notes = ref([])
  const note = ref('')
  
  const handleNote = ()=>{
    notes.value.push(note.value)
    note.value = ''
  }
  </script>
  <!-- ************************************* -->
  <style scoped>
  .container{
    margin: 2em;
    padding: 2em;
    border: brown 4px double;
  }
  
  .list-enter-active, .list-leave-active{
    transition: all 1s ease;
  }
  .list-enter-from, .list-leave-to{
    opacity: 0;
    transform:  translateX(30px);
  }
  </style>
