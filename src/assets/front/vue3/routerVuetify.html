<template>
<div>
<v-app>
  <v-app-bar class="bg-blue">
    <v-toolbar-title>Mi App</v-toolbar-title>
    <v-spacer></v-spacer>
    <v-btn
    v-for="(route, index) in routes" :key="index"
    @click="navigate(route.path)"
    :color="isActive(route.path)? 'purple' : 'white'"
    text
    >{{ route.name }}</v-btn>
  </v-app-bar>
  <v-main >
    <router-view/>
  </v-main>
</v-app>
</div>
</template>
<!-- ************************************* -->
<script setup>
import {useRouter, useRoute  } from "vue-router";

let router = useRouter()
let route = useRoute()

const routes = [
{ name: 'Home', path: '/' },
{ name: 'About', path: '/about' },
{ name: 'Practica', path: '/practica' }
]

const navigate = (path) => {
router.push(path)
}

let isActive = (path) => {
return route.path ===path
}

</script>
