            <template>
            <div>
             <h1>Practica Uno</h1>  <br><br>
              <h2>Ahorros  $.{{ ahorro }}</h2> <br><br>
              <h2>Cantidad de movimientos {{ movimiento }}</h2> <br><br>
              <v-btn @click="deposito" class="btn">Depósito</v-btn>
              <v-btn @click="retiro">Retiro</v-btn> <br><br><br>
           
              <h3>MOVIMIENTOS </h3>
              <ol>
                <li v-for="(mov, index) in movimientos" :key="index">{{ mov }}</li>
              </ol>
            </div>
          </template>
          <!-- ******************************* --> 
          <script setup>
          import { ref } from 'vue';
          
          
          let ahorro = ref(0)
          let movimiento = ref(0)
          let movimientos = ref([])
          
          let retiro = ()=>{
            if (ahorro.value > 0) {
              ahorro.value -= 100
              movimiento.value++
              movimientos.value.push('Retiro $100.00')
              
            }else{
              alert('No tiene Fondos')
            }
          }
          let deposito = ()=>{
            ahorro.value += 100
            movimiento.value++
            movimientos.value.push('Deposito $100.00')
          }
          </script>
