Reactividad

  1. botón que incremente de 100 en 100
  2. Que cambie de color pasando 1000
  3. inicie en 500
///////////////////////////////////////////////////////// 
    

    
<template>
<div>
 <h1>REACTIVIDAD</h1> 
  <h2 :class="ahorro>1000 ? 'blue' :'green'">Ahorro : {{ ahorro }}</h2> <br><br>

  <v-btn @click="aumento">Ingresar $ 100</v-btn>
</div>
</template>
<!-- ******************************* --> 
<script setup>
import { ref } from 'vue';

let ahorro = ref(500)
let aumento = () => {
ahorro.value+= 100
}
</script>
<!-- ******************************* -->
<style scoped>
.blue{
color: blue;
}
.green{
color: green;
}
div{
background-color: blanchedalmond;
}
</style>
