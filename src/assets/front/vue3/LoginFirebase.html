      <template>
  <div class="container">
      <h1>Auth</h1>
      <form>
  <div class="mb-3">
    <label for="example" >Correo</label>
    <input type="email" class="form-control" id="example" aria-describedby="emailHelp" v-model="email">
  </div>

  <div class="mb-3">
    <label for="example" >Contraseña</label>
    <input type="password" class="form-control" id="example" v-model="pass">
  </div>

  <button type="submit" class="btn btn-primary" @click.prevent="authUser">Submit</button>

</form>
  </div>
</template>
<!-- ************************************* -->
<script setup>
import {ref} from 'vue'
import {signInWithEmailAndPassword, getAuth} from 'firebase/auth'
let email = ref("")
let pass = ref("")

const authUser = () =>{
  const auth = getAuth()
  signInWithEmailAndPassword(auth, email.value, pass.value).then(()=>{
    alert('Muy bien')
  })
    .catch((error) =>{
      alert("error: " + error.message)
    })
  }

</script>
