FUNCIONES

  1. Llamar un funcion con un nombre
  2. una con un saludo
  3. usar un return en la funcion anterior
  4. usar parámetros
  5. valores por defecto en la anterior
  6. asignar la función a una variable
    


      //FUNCIÓN DECLARADA. Se puede invocar incluso antes de ser declarada
      //sin parametros
      function nombre() {
         console.log('hola mundo');
      }
      nombre()
      
      // que devuelve valor
      // uso de return
      function saludo(){
        console.log('buenos días');
        return'valor de funcion'
      }
      let saludan = saludo()
      console.log(saludan)
      
      // con parámetros
      function saludar (nombre, edad){
        console.log(`Mi nombre es ${nombre} y mi edad es ${edad} años`)
      }
      saludar('Carlos', 65)
      
      //valores por defecto
      function pedido (producto='ninguno', cantidad=0){
        console.log(`El cliente requiere el pruducto: ${producto} y la cantidad es: ${cantidad}`);
      }
      pedido()
      
      //FUNCION EXPRESADA Solo se puede invocar después de declararse
      //Es la más recomendada para mejor estructuración del código
      // ASIGNAR VALOR A UNA VARIABLE(función anónima)
      let tarea = function(){
        console.log('Tarea iniciada');
      }
      tarea()
