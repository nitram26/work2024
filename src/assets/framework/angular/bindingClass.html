            TS. 

            import { Component } from '@angular/core';

            @Component({
              selector: 'app-accion',
              standalone: true,
              imports: [],
              templateUrl: './accion.component.html',
              styleUrl: './accion.component.css'
            })
            export class AccionComponent {

            color: string = ''
            cambio : boolean = false

            uno (){
              this.color = 'green'
            }
            dos (){
              this.color = 'red'
            }
            tres (){
              this.color = 'blue'
            }

            deshabilitar (){
              this.cambio = !this.cambio
            }

            }
