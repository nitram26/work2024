Binding con Clases 2159

  1. Tres botones cambien el color del texto
  2. luego un botón que habilite y deshabilite los botones
  3. El click se agrega entre parentesis
  4. el disabled se pone [disabled] = "nombreDeVariable"
  5. las clases para css se agregan solo con la palabra class
          

            HTML. 

            <h1>BINDING</h1>
            <button 
            [disabled]="cambio"
            (click)="uno()">color1</button>

            <button 
            [disabled]="cambio"
            (click)="dos()">
            color2</button>

            <button 
            [disabled]="cambio" 
            (click)="tres()">
            color3</button>
            
            <span [class]="color">Este texto cambiará de color</span>
            
            <button 
            (click)="deshabilitar()">hab/deshab</button>

            {{cambio}}

            TS. 

            import { Component } from '@angular/core';

            @Component({
              selector: 'app-accion',
              standalone: true,
              imports: [],
              templateUrl: './accion.component.html',
              styleUrl: './accion.component.css'
            })
            export class AccionComponent {

            color: string = ''
            cambio : boolean = false

            uno (){
              this.color = 'green'
            }
            dos (){
              this.color = 'red'
            }
            tres (){
              this.color = 'blue'
            }

            deshabilitar (){
              this.cambio = !this.cambio
            }

            }
