CONDICIONAL IF 1044

  1. Un input que muestre por consola lo escrito
  2. Esto a traves de un botón
  3. Una condicional que muestre SE HA ENVIADO MENSAJE al pulsar el botón
  4. Un botón cerrar
  5. Y un else que muestre Mensaje de prueba mientras no se envié
    6. NOTA IMPORTANTE
  6. Para usar las condicionales se tiene que importar el CommonModule
  7. Tanto en el appModules.ts como en el ts del componente

AGREGAR EL COMMON MODULE EN APPMODULES.TS

    

      
TEMPLATE

      <input type="text" #name>

<button (click)="cambio(name.value)">Press</button>

<h4 *ngIf="mensaje"> {{name.value}} </h4>

<p *ngIf="mensaje">el mensaje ha sido enviado</p>



SCRIPT

import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';


@Component({
  selector: 'app-aarepasar',
  standalone: true,
  imports: [CommonModule],
  templateUrl: './aarepasar.component.html',
  styleUrl: './aarepasar.component.css'
})
export class AarepasarComponent {

  mensaje : boolean = false
cambio(name: string) {
  name
  this.mensaje=true
  setTimeout(() => {
    this.mensaje=false

  }, 3000);
}   
}
