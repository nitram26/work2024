EMITS

  1. Hijo: crear una función en el template
  2. Definir el emit y la función con el nombre y elementos de cambios
  3. Padre: se enlaza la función del hijo con una función del padre 
          

            
          
PADRE

<template>
  <v-container :fluid="true">
   <h1>Padre</h1> <br><br>

   <WorkView :nombre="nombre" :edad="edad" @modificar="cambiarValor"/>
  </v-container>
</template>
<!-- ******************************* --> 
<script setup>
import { ref } from 'vue'
import WorkView from './WorkView.vue'

const cambiarValor=(valor1, valor2)=>{
  nombre.value = valor1
  edad.value = valor2
}
let nombre = ref('Carlos')
let edad = ref(65)
</script>
<!-- ******************************* -->
<style scoped>

</style>

HIJO


<template>
  <div>
   <h1>Hijo</h1> 
   <h2>Te llamas {{ nombre }} y tienes {{ edad }} años</h2> <br><br>

   <v-btn @click="enviar">Cambiar valores</v-btn>
  </div>
</template>
<!-- ******************************* --> 
<script setup>

let emit = defineEmits(['modificar'])
const enviar=()=>{
  emit('modificar', 'Martin', 38)
}

const props = defineProps({
  nombre: String,
  edad: Number
})
</script>
<!-- ******************************* -->
<style scoped>
div{
  background-color: blanchedalmond;
}
</style>
