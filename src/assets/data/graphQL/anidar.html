      import {gql, ApolloServer} from 'apollo-server'


      const persons = [
        {
          name: "Carlos",
          phone: "102030",
          street: "Las Palmeras 74",
          city: "Lima",
          age:"65",
          id: "1",
        },
        {
          name: "Sofia",
          phone: "405060",
          street: "Las Palmeras 75",
          city: "Pucusana",
          age:"47",
          id: "2",
        },
        {
          name: "Rebeca",
          phone: "708090",
          street: "Las Palmeras 76",
          city: "Huancayo",
          age:"30",
          id: "3",
        },
        {
          name: "Adrian",
          phone: "321654",
          street: "Las Palmeras 76",
          city: "Ayacucho",
          age:"14",
          id: "4",
        },
      ];

      const typeDefs = gql`
      type Address {
        street:String!
        city: String
      }

      type Person {
        name: String!
          phone: String
          address: Address!
          age:String
          id: ID!
          canDrink: Boolean
      }
        type Query {
          personCount: Int!
          allPersons : [Person]!
          findPerson(name: String!): Person
        }
      `

      const resolvers = {
        Query:{
        personCount: () => persons.length,
        allPersons : () => persons,
        findPerson : (root, args) => {
          const {name} = args
          return persons.find(person => person.name === name)
          }
        },
        Person: {
          address : (root) =>{
           return{
            street: root.street,
            city: root.city
           }
          }
        }
      }

      const server = new ApolloServer({
        typeDefs,
        resolvers
      })

      server.listen().then(({url}) => {
        console.log(`Server ready at ${url}`);
      })
