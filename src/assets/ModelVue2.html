'use strict'

const mongoose = require('mongoose')
const Schema = mongoose.Schema

const InnovationSchema = Schema({
  clientName: {
    type: String,
  },
  title: {
    type: String,
  },
  description: {
    type: String,
  },
})

module.exports = mongoose.model('Innovations', InnovationSchema)
